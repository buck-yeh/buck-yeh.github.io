323const C_NFA<T_Inputs,T_Action,C_Traits> &C_NFA<T_Inputs,T_Action,C_Traits>::checkReuse(
324 const std::set<C_NfaState> &usedStates,
325 C_NFA<T_Inputs,T_Action,C_Traits> &replacement ) const
326{
327 typedef std::map<C_NfaState,C_NfaState> C_Map;
328 C_Map map;
329 for (const auto &i: delta)
330 {
331 if (usedStates.find(i.first) != usedStates.end())
332 map[i.first];
333
334 for (const auto &j: i.second)
335 if (usedStates.find(j.first) != usedStates.end())
336 map[j.first];
337 }
338 if (map.empty())
339 // No conflict
340 return *this;
341
342 C_Map::const_iterator found = map.find(q0);
343 replacement.q0 = found != map.end()? found->second: q0;
344
345 for (const auto &i: F)
346 {
347 found = map.find(i);
348 replacement.F.emplace_back(found != map.end()? found->second: i, i.m_Tag);
349 }
350
351 for (const auto &i: delta)
352 {
353 found = map.find(i.first);
354 auto &t =replacement.delta[found != map.end()? found->second: i.first];
355 for (const auto &j: i.second)
356 {
357 found = map.find(j.first);
358 t[found != map.end()? found->second: j.first] =j.second;
359 }
360 }

445void C_DFA<T_Inputs,T_Action,C_Traits>::minDfa(
446 const C_FinalMap &Ffat,
447 const C_TransitionMap &deltaFat,
448 int totalFatStates,
449 C_FinalMap &Fmin,
450 C_TransitionMap &deltaMin )

452 // Figure 7.7: Algorithm MinDFA
453 C_DfaClosures Qfat;
454 Qfat.emplace_back();
455 Qfat.emplace_back();
456 auto &part0 = Qfat.front().m_Closure;
457 auto &part1 = Qfat.back().m_Closure;
458 int tag =0;
459 for (const auto &i: Ffat)
460 {
461 part0.insert(i.first);
462 while (tag < i.first)
463 part1.insert(tag++);
464
465 ++tag;
466 }
467 while (tag < totalFatStates)
468 part1.insert(tag++);
469
470 bool changed;
471 do
472 {
473 changed = false;
474 C_DfaClosures Qmin;
475 for (const auto &i: Qfat)
476 {
477 auto fat = i.m_Closure;
478 while (!fat.empty())
479 {
480 C_DfaClosure slim;
481 auto j = fat.begin();
482 slim.insert(*j);
483 typedef std::map<const C_DfaClosure*,T_Inputs> C_CmpMap;
484 C_CmpMap cmpmap;
485 auto k = deltaFat.find(*j);
486 if (k != deltaFat.end())
487 for (const auto &m: k->second)
488 cmpmap[&findDfaClosure(Qfat, m.first)->m_Closure] = m.second;
489
490 while (++j != fat.end())
491 {
492 C_CmpMap map;
493 bool deal = true;
494 k = deltaFat.find(*j);
495 if (k != deltaFat.end())
496 for (const auto &m: k->second)
497 {
498 const auto *t = &findDfaClosure(Qfat, m.first)->m_Closure;
499 auto found = cmpmap.find(t);
500 if (found != cmpmap.end() &&
501 C_Traits::equalInput(m.second, found->second))
502 {
503 map[t] = m.second;
504 continue;
505 }
506 deal = false;
507 break;
508 }
509 if (deal)
510 {
511 auto i_map = map.begin();
512 for (auto &j_cmp: cmpmap)
513 {
514 if (i_map == map.end() ||
515 i_map->first != j_cmp.first ||
516 !C_Traits::equalInput(i_map->second, j_cmp.second))
517 {
518 deal = false;
519 break;
520 }
521 ++i_map;
522 }
523 if (deal && i_map == map.end())
524 slim.insert(*j);
525 }
526 }
527 Qmin.emplace_back(slim);
528 C_DfaClosure temp;
529 std::set_difference(fat.begin(), fat.end(),
530 slim.begin(), slim.end(), std::inserter(temp, temp.begin()));
531 fat.swap(temp);
532 }
533 }
534 if (Qmin.size() == Qfat.size())
535 {
536 for (auto i =Qmin.begin(), end =Qmin.end(); i != end; ++i)
537 {
538 C_Action2Closure a2c;
539 buildA2C(Ffat, i->m_Closure, a2c);
540 if (a2c.size() > 1)
541 {
542 // Solve conflict
543 auto j = a2c.begin();
544 i->m_Closure.swap(j->second); // overwrite the superset
545 auto next = i;
546 ++next;
547 while (++j != a2c.end())
548 i = Qmin.insert(next, j->second);
549 }
550 }
551 }
552 if (Qmin.size() != Qfat.size())
553 {
554 Qfat.swap(Qmin);
555 changed =true;
556 }
557 } while (changed);
558
559 // Tagging
560 tag =0;
561 for (auto i = Qfat.begin(), end = Qfat.end(); i != end; ++i, ++tag)
562 {
563 if (tag && i->m_Closure.find(0) != i->m_Closure.end())
564 // Q0 is always tagged 0
565 {
566 i->m_Tag = 0;
567 Qfat.front().m_Tag = tag;
568 auto t = Fmin.find(0);
569 if (t != Fmin.end())
570 {
571 const auto a = t->second;
572 Fmin.erase(t);
573 Fmin[tag] = a;
574 }
575 }
576 else
577 i->m_Tag = tag;
578
579 C_Action2Closure a2c;
580 buildA2C(Ffat, i->m_Closure, a2c);
581 if (!a2c.empty())
582 {
583 Fmin[i->m_Tag] = a2c.begin()->first;
584 if (a2c.size() > 1)
585 // Impossible
586 RUNTIME_ERROR("a2c.size() == ()", a2c.size());
587 }
588 }
589 for (const auto &i: deltaFat)
590 {
591 auto &dst_i = deltaMin[findDfaClosure(Qfat,i.first)->m_Tag];
592 for (const auto &j: i.second)
593 {
594 const auto value = findDfaClosure(Qfat,j.first)->m_Tag;
595 for (const auto &dst_j: dst_i)

604 }

606 C_Traits::inputUnion(dst_i[value], j.second);
607 PostInsertion:;
608 }

613template<class F_PickAction>
614int C_DFA<T_Inputs,T_Action,C_Traits>::nfa2dfa(const C_NFA<T_Inputs,T_Action,C_Traits> &nfa,
615 C_FinalMap &F, C_TransitionMap &delta, F_PickAction pickAction)
616{
617 // Figure 7.5 (p245): Algorithm NFA->DFA
618 std::list<C_TaggedNfaClosure> Q; // collector of all states
619 Q.emplace_back(C_NfaClosure{});
620 auto *const q0 = &Q.front().m_Closure; // starting point index of Q
621 q0->insert(nfa.q0);
622 createClosure(*q0, nfa.delta);
623
624 typedef std::pair<T_Inputs, const C_TaggedNfaClosure*> C_InputClosurePair;
625 typedef std::vector<C_InputClosurePair> C_Input2Closure;
626 std::map<const C_TaggedNfaClosure*, C_Input2Closure,
627 bool (*)(const C_TaggedNfaClosure*,const C_TaggedNfaClosure*)>
628 deltaNfa([](const C_TaggedNfaClosure *a, const C_TaggedNfaClosure *b)->bool {
629 return a->m_Closure < b->m_Closure;
630 }); // transition relation
631 bool changed;
632 do
633 {
634 changed =false;
635 for (auto &q: Q)
636 {
637 if (q.m_Tag)
638 continue;
639
640 typedef std::map<C_NfaClosure,T_Inputs> C_ShiftMap;
641 C_ShiftMap map;
642 q.m_Tag = true;
643 for (const auto &j: q.m_Closure)
644 {
645 auto found = nfa.delta.find(j);
646 if (found != nfa.delta.end())
647 for (const auto &k: found->second)
648 {
649 if (!C_Traits::isEmptyInput(k.second.m_UserInputs))
650 {
651 C_NfaClosure key;
652 key.insert(k.first);
653 C_Traits::inputUnion(map[key], k.second.m_UserInputs);
654 }
655 }
656 }
657 // Here map is already created with keys of state singletons and values without epsilon
658 C_ShiftMap partitionedMap;
659 MergeAgain:
660 switch (map.size())
661 {
662 case 1:
663 partitionedMap.insert(*map.begin());
664 break;
665 case 0:
666 break;
667 default: // > 1
668 for (auto i = map.begin(), end = map.end(); i != end; ++i)
669 for (auto j = i; ++j != end;)
670 {
671 // Emptyness of (i - j)
672 T_Inputs insetsect(i->second);
673 C_Traits::inputIntersection(insetsect, j->second);
674 if (C_Traits::isEmptyInput(insetsect))
675 continue;
676
677 if (map.begin() != i)
678 // Move away the disjointed part first
679 {
680 partitionedMap.insert(map.begin(), i);
681 map.erase(map.begin(), i);
682 }
683 C_NfaClosure key(i->first);
684 key.insert(j->first.begin(), j->first.end());
685
686 T_Inputs t(i->second);
687 C_Traits::inputDifference(t, insetsect);
688 if (C_Traits::isEmptyInput(t))
689 map.erase(i);
690 else
691 i->second = t;
692
693 t = j->second;
694 C_Traits::inputDifference(t, insetsect);
695 if (C_Traits::isEmptyInput(t))
696 map.erase(j);
697 else
698 j->second = t;
699
700 map[key] =insetsect;
701 goto MergeAgain;
702 }
703 partitionedMap.insert(map.begin(), map.end());
704 } // switch (map.size())
705 map.swap(partitionedMap);
706 /* Here map is with
707 * keys union of which equals to union of original keys

739 conflict.insert(j.m_Tag);

741 if (!conflict.empty())

743 if (conflict.size() > 1)

747 if (action == T_Action())
748 RUNTIME_ERROR("Interrupted");

750 F[i.m_Tag] = action;
751 }

753 F[i.m_Tag] = *conflict.begin();
754 }
