95int u32toutf8(T_Utf32 c, T_Utf8 *dst) noexcept
96{
97 int ret;
98 if (c < 0x80)
99 // 1 byte
100 {
101 ret =1;
102 *dst = T_Utf8(c);
103 }
104 else if (c < 0x800)
105 // 2 bytes
106 {
107 ret =2;
108 goto Encode;
109 }
110 else if (c < 0x10000)
111 // 3 bytes
112 {
113 ret =3;
114 goto Encode;
115 }
116 else if (c < 0x200000)
117 // 4 bytes
118 {
119 ret =4;
120 goto Encode;
121 }
122 else if (c < 0x4000000)
123 // 5 bytes
124 {
125 ret =5;
126 goto Encode;
127 }
128 else if (c < 0x80000000)
129 // 6 bytes
130 {
131 ret =6;
132 goto Encode;
133 }
134 else
135 ret =-1;
136
137 return ret;
138Encode:
139 for (int i =ret; --i > 0;)
140 {
141 dst[i] = T_Utf8((c &0x3F) | 0x80);
142 c >>=6;
143 }
144 dst[0] = T_Utf8(c |((const unsigned char*)"\xC0\xE0\xF0\xF8\xFC")[ret-2]);
145 return ret;
146}

329void C_UnicodeIn::init()
330{
331 m_Src.read(2);
332 switch (m_Src.size())
333 {
334 case 2:
335 switch (m_Src.getUtf16(0))
336 {
337 case 0xFEFF: // UTF-16 with BOM
338 m_Src.pop(2);
339 m_ReadMethod = &C_UnicodeIn::readUTF16;
340 return;
341 case 0xFFFE: // Reverse UTF-16 with BOM
342 m_Src.pop(2);
343 m_ReadMethod = &C_UnicodeIn::readReverseUTF16;
344 return;
345 default:
346 m_Src.read(3);
347 if (m_Src.size() >= 3 && 0 == memcmp(m_Src.buffer(), "\xef\xbb\xbf", 3))
348 // UTF-8 with BOM
349 {
350 m_Src.pop(3);
351 setCodePage(CHSETS_UTF8);
352 m_ReadMethod =&C_UnicodeIn::readCodePage;
353 return;
354 }
355
356 // Infer the encoding of the 1000-byte header chunk (UTF-8, ACP, or something else ?)
357 if (!m_CodePage)
358 {
359 m_Src.read(1000);
360 const auto size = m_Src.size();
361#ifdef _WIN32
362 int mask = IS_TEXT_UNICODE_UNICODE_MASK;
363 if (IsTextUnicode(m_Src.buffer(), int(size), &mask))
364#elif defined(__unix__)
365 if (testUtf16(m_Src.buffer(), size))
366#endif
367 {
368 m_ReadMethod =&C_UnicodeIn::readUTF16;
369 return;
370 }
371 const auto revBuf = std::make_unique<char[]>(size);
372 swab(const_cast<char*>(m_Src.buffer()), revBuf.get(), int(size));
373#ifdef _WIN32
374 mask = IS_TEXT_UNICODE_UNICODE_MASK;
375 if (IsTextUnicode(revBuf.get(), int(size), &mask))
376#elif defined(__unix__)
377 if (testUtf16(revBuf.get(), size))
378#endif
379 {
380 m_ReadMethod =&C_UnicodeIn::readReverseUTF16;
381 return;
382 }
383 }
384
385 // Read heading asciis off then we can distinguish CP_UTF8 from CP_ACP.
386 m_ReadMethod = &C_UnicodeIn::readASCII;
387 }
388 break;
389 case 1: // has to be ascii
390 if (*m_Src.buffer() &0x80)
391 m_ErrCode = UIE_INCOMPLETE_UNICODE;
392 else
393 m_GetQ.push(T_Utf8(*m_Src.buffer()));
394 break;
395 case 0: // empty string
396 break;
397 default:
398 m_ErrCode = UIE_INTERNAL;
399 }

455bool C_UnicodeIn::readUTF16(C_Source &src, bool reverseWord)
456{
457 bool ret = false;
458 src.read(2);
459 const size_t read =src.size();
460 if (read >= 2)
461 {
462 const T_Utf16 uc = src.getUtf16(0,reverseWord);
463 if (0xD800 <= uc && uc < 0xDC00)
464 // Hi word of 2-word encoding
465 {
466 src.read(4);
467 if (src.size() >= 4)
468 {
469 const T_Utf16 uc2 = src.getUtf16(1,reverseWord);
470 if (0xDC00 <= uc2 && uc2 < 0xE000)
471 // Low word of 2-word encoding
472 {
473 src.pop(4);
474 m_GetQ.push(T_Utf32((((uc&0x3FF)<<10)|(uc2&0x3FF))+0x10000));
475 ret =true;
476 }
477 else
478 // Anything lese is ill-formed
479 m_ErrCode = UIE_ILLFORMED_UNICODE;
480 }
481 else
482 m_ErrCode = UIE_INCOMPLETE_UNICODE;
483 }
484 else if (0xDC00 <= uc && uc < 0xE000)
485 // Low word of 2-word encoding - ill-fomed
486 m_ErrCode = UIE_ILLFORMED_UNICODE;
487 else
488 {
489 src.pop(2);
490 m_GetQ.push(uc);
491 ret = true;
492 }
493 }
494 else if (read > 0)
495 m_ErrCode = UIE_INCOMPLETE_UNICODE;
496
497 return ret;
498}

721const std::string &C_MBCStr::escJSON() const
722{
723 return escape([](std::string &dst, char c) {
724 /* From http://stackoverflow.com/questions/19176024/how-to-escape-special-characters-in-building-a-json-string
725 \b Backspace (ascii code 08)
726 \f Form feed (ascii code 0C)
727 \n New line
728 \r Carriage return
729 \t Tab
730 \" Double quote
731 \\ Backslash character
732 */
733 static constexpr char JSON_ESCS[] = "\b\f\n\r\t\"\\";
734 if (auto p = strchr(JSON_ESCS, c))
735 {
736 dst += '\\';
737 dst += "bfnrt\"\\"[p - JSON_ESCS];
738 }
739 else if (isprint(c))
740 dst += c;
741 else
742 dst += fmt::format("\\u{:04}", int{static_cast<unsigned char>(c)});
743 });
