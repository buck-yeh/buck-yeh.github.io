455 {
456  bool ret = false;
457  src.read(2);
458  const size_t read =src.size();
459  if (read >= 2)
460  {
461  const T_Utf16 uc = src.getUtf16(0,reverseWord);
462  if (0xD800 <= uc && uc < 0xDC00)
463  // Hi word of 2-word encoding
464  {
465  src.read(4);
466  if (src.size() >= 4)
467  {
468  const T_Utf16 uc2 = src.getUtf16(1,reverseWord);
469  if (0xDC00 <= uc2 && uc2 < 0xE000)
470  // Low word of 2-word encoding
471  {
472  src.pop(4);
473  m_GetQ.push(T_Utf32((((uc&0x3FF)<<10)|(uc2&0x3FF))+0x10000));
474  ret =true;
475  }
476  else
477  // Anything lese is ill-formed
478  m_ErrCode = UIE_ILLFORMED_UNICODE;
479  }
480  else
481  m_ErrCode = UIE_INCOMPLETE_UNICODE;

484  // Low word of 2-word encoding - ill-fomed
