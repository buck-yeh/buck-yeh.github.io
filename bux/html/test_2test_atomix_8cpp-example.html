TEST_CASE("Empty cache", "[Z]")
{
bux::C_SpinCacheT<int,std::string> cache;
REQUIRE(cache.size() == 0);
}
TEST_CASE("Cache for one thread", "[O]")
{
bux::C_SpinCacheT<int,std::string> cache;
for (int i = 0; i < 100; ++i)
{
const int t = i % 10;
cache(t, [t](std::string &s){ s = std::to_string(t); });
}
REQUIRE(cache.size() == 10);
}
