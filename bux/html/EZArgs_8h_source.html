193{

214 const std::ranges::forward_range auto &count_optionals,
215 bool unlimited )
226{
227 if (!std::empty(arg_names) || unlimited)
228 // Non-trivial case
229 {
230 switch (m_up2u.index())
231 {
232 case 0: // turn case 1
233 m_up2u.emplace<C_ArgLayout>();

241 dst.m_posArgs.assign(std::begin(arg_names), std::end(arg_names));
242 dst.m_unlimited = unlimited;

245 dst.m_posCounts.assign(std::begin(count_optionals), std::end(count_optionals));
246 sort(dst.m_posCounts.begin(), dst.m_posCounts.end());
247 const auto n_args = dst.m_posArgs.size();
248 while (!dst.m_posCounts.empty() && dst.m_posCounts.back() >= n_args)
249 dst.m_posCounts.pop_back();
250
251 dst.m_posCounts.emplace_back(n_args);
252 }
253 return *this;

256C_ErrorOrIndex C_EZArgs::parse(std::integral auto argc, const char *const argv[]) const
263{
264 decltype(argc) ind = 1;
265 switch (m_up2u.index())
266 {
267 case 0: // trivial case
268 break;
269 case 1: // with positional arguments
270 for (; ind < argc && argv[ind][0] != '-'; ++ind);
271 m_totoalPositionArgs = ind;
272 if (ind < argc && (argv[ind][1] == 'h' || !strcmp(argv[ind], "--help")))
273 return help_full(argv);
274 else
275 {
276 auto &lo = std::get<1>(m_up2u);
277 decltype(argc) lower_bound = 0;
278 for (auto i: lo.m_posCounts)
279 {
280 if (lower_bound < ind && ind <= static_cast<decltype(argc)>(i))
281 {
282 std::string message;
283 if (!lower_bound)
284 {
285 const bool plural = i > 1;
286 message = "Positional argument";
287 if (plural)
288 message += 's';
289 for (size_t j = 0; j < i; ++j)
290 message.append(" <").append(lo.m_posArgs[j]) += '>';
291
292 message += plural? " are required": " is required";
293 }
294 else
295 message = std::to_string(ind-1).append(ind > 2? " positonal arguments are": " positonal argument is").append(" not allowed");
296
297 return {help_tip(message,argv), ind};
298 }
299 lower_bound = static_cast<decltype(argc)>(i + 1);
300 }
301 if (!lo.m_unlimited && ind > static_cast<decltype(argc)>(lo.m_posArgs.size() + 1))
302 {
303 std::string message = "Too many positional arguments:";
304 for (auto &i: lo.m_posArgs)
305 message.append(" <").append(i) += '>';
306 return {help_tip(message,argv), ind};
307 }
308 }
309 break;
310 case 2: // into subcommands
311 if (ind >= argc || !strcmp(argv[ind],"-h") || !strcmp(argv[ind],"--help"))
312 return help_full(argv);
313 else
314 {
315 auto &map = std::get<2>(m_up2u);
316 auto i = map.find(argv[ind]);
317 if (i == map.end())
318 return {help_tip(std::string("Unknown subcommand: ")+argv[ind], argv), ind};
319
320 auto ret = i->second.parse(argc-ind, argv+ind);
321 if (ret)
322 {
323 m_totoalPositionArgs = i->second.m_totoalPositionArgs + ind;
324 return ret.m_optIndex.value() + ind;
325 }
326 return ret;
327 }
328 }
329 const auto flagStartInd = ind;
330
331 // Parse un-ordered flags
332 for (; ind < argc; ++ind)
333 {
334 auto arg = argv[ind];
335 if (arg[0] != '-')
336 // Not flag initial
337 return {help_tip(std::string("Unexpected argument: ")+arg, argv), ind};
338
339 if (arg[1] == '-')
340 // Match full flag name
341 {
342 const auto flag = arg + 2;
343 const auto eqsign = strchr(flag, '=');
344 const auto flag_name = eqsign? std::string_view(flag, size_t(eqsign-flag)): std::string_view(flag);
345 for (auto cur = this; cur; cur = cur->m_owner)
346 for (auto &def: cur->m_flags)
347 if (def.m_name == flag_name)
348 // Matched
349 {
350 if (eqsign)
351 // (=)-connected flag value
352 {
353 if (def.m_parse)
354 {
355 def.m_parse(eqsign+1);
356 goto NextInd;
357 }
358 return {help_tip(std::string("Value parser absent: ")+arg, argv), ind};
359 }
360 else if (ind+1 < argc && argv[ind+1][0] != '-')
361 // Flag with value
362 {
363 if (def.m_parse)
364 {
365 def.m_parse(argv[++ind]);
366 goto NextInd;
367 }
368 return {help_tip(std::string("Value parser absent: ")+arg, argv), ind};
369 }
370 else
371 // Flag without value
372 {
373 if (def.m_trigger)
374 {
375 def.m_trigger();
376 goto NextInd;
377 }
378 return {help_tip(std::string(def.m_parse?"Missing flag value: ":"Triggerless flag: ")+arg, argv), ind};
379 }
380 }
381 if ("help" == flag_name)
382 return help_full(argv);
383 else
384 return {help_tip(std::string("Unknown flag: --")+=flag_name, argv), ind};
385 } // if (arg[1] == '-')
386 else
387 // Match continous short flag names
388 while (char sname = *++arg)
389 {
390 for (auto cur = this; cur; cur = cur->m_owner)
391 for (auto &def: cur->m_flags)
392 if (def.m_shortName == sname)
393 {
394 if (!arg[1] && ind+1 < argc && argv[ind+1][0] != '-')
395 // Flag with value
396 {
397 if (def.m_parse)
398 {
399 def.m_parse(argv[++ind]);
400 goto NextInd;
401 }
402 return {help_tip(std::string("Value parser absent: -")+sname, argv), ind};
403 }
404 else
405 // Flag without value
406 {
407 if (def.m_trigger)
408 {
409 def.m_trigger();
410 goto NextInd;
411 }
412 return {help_tip(std::string(def.m_parse?"Missing flag value: -":"Triggerless flag: -")+sname, argv), ind};
413 }
414 }
415 if ('h' == sname)
416 return help_full(argv);
417 else
418 return {help_tip(std::string("Unknown flag: -")+=sname, argv), ind};
419 } // while (char sname = *++arg)
420 NextInd:;
421 } // for (; ind < argc; ++ind)
422
423 if (m_onParsed)
424 m_onParsed();
425
426 return flagStartInd;
427}
