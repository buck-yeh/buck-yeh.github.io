126template<class C1, class C2>
127C_EZArgs &C_EZArgs::position_args(const C1 &arg_names, const C2 &count_optionals, bool unlimited)
128{
129 if (!std::empty(arg_names) || unlimited)
130 // Non-trivial case
131 {
132 switch (m_up2u.index())
133 {
134 case 0: // turn case 1
135 m_up2u.emplace<C_ArgLayout>();
136 break;
137 case 1:
138 break;
139 case 2:
140 RUNTIME_ERROR("Already added subcommands");
141 }
142 auto &dst = std::get<1>(m_up2u);
143 dst.m_posArgs.assign(std::begin(arg_names), std::end(arg_names));
144 dst.m_unlimited = unlimited;
145
146 // Assign & adjust m_posCounts
147 dst.m_posCounts.assign(std::begin(count_optionals), std::end(count_optionals));
148 sort(dst.m_posCounts.begin(), dst.m_posCounts.end());
149 const auto n_args = dst.m_posArgs.size();
150 while (!dst.m_posCounts.empty() && dst.m_posCounts.back() >= n_args)
151 dst.m_posCounts.pop_back();
152
153 dst.m_posCounts.emplace_back(n_args);
154 }
155 return *this;
156}
