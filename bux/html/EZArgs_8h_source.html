143  dst.m_posArgs.assign(std::begin(arg_names), std::end(arg_names));
144  dst.m_unlimited = unlimited;
145 
146  // Assign & adjust m_posCounts
147  dst.m_posCounts.assign(std::begin(count_optionals), std::end(count_optionals));
148  sort(dst.m_posCounts.begin(), dst.m_posCounts.end());
149  const auto n_args = dst.m_posArgs.size();
150  while (!dst.m_posCounts.empty() && dst.m_posCounts.back() >= n_args)
151  dst.m_posCounts.pop_back();
