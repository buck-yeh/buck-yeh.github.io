418 {
419  if (m_ArrSize >= m_ArrLimit)
420  {
421  const size_t limit =m_ArrLimit? m_ArrLimit *2: m_MinAlloc;
422  T *const rebuf =reinterpret_cast<T*>(new char[limit*sizeof(T[1])]);
423  size_t i;
424  for (i =0; i < m_ArrSize; ++i)
425  new(rebuf+i) T(m_RawSpace[i]);
