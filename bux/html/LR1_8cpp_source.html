212 {
213  bool isLast = true;
214  for (auto i = m_CurStack.size(); i; --i, isLast = false)
215  {
216  if (!isLast && m_CurStack[i].m_TokenID == m_Policy.m_IdError)
217  // No crossing old m_Policy.m_IdError
218  break;
219 
220  if (m_Policy.action(m_CurStack[i-1].m_StateID, m_Policy.m_IdError) != ACTION_ERROR)
221  {
222  if (!isLast && info.m_pos == m_CurStack[i].m_pos)
223  // Total length of nonempty ungotten tokens is zero.
224  continue;
